/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode(int x) { val = x; }
 * }
 */
public class Solution {
    boolean f=true;
    public boolean isBalanced(TreeNode root) {
        if(root==null){
            return true;
        }
        if(root.left==null&&root.right==null){
            return true;
        }
        depth(root);
        return f;
    }
    public int depth(TreeNode root){
        if(root.left==null&&root.right==null){
            return 1;
        }
        int ldepth=1,rdepth=1;
        if(root.left!=null){
            ldepth = depth(root.left)+1;
        }
        if(root.right!=null){
            rdepth = depth(root.right)+1;
        }
        if(ldepth-rdepth>1||rdepth-ldepth>1){
            f=false;
        }
        if(ldepth>=rdepth){
            return ldepth;
        }
        else{
            return rdepth;
        }
    }
}
